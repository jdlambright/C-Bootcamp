7 student averages  had to put (decimal) at front of calculation to make sure that it would not round.
using System;

namespace firstConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {

            string[] studentNames = { "fred", "barbara", "sally", "jake", "amy"};
            int[,] studentGrades = new int[5, 4];
            decimal[] studentAverage = new decimal[5];

            //fred grades
            studentGrades[0, 0] = 98;
            studentGrades[0, 1] = 86;
            studentGrades[0, 2] = 89;
            studentGrades[0, 3] = 94;

            //barabara grades 
            studentGrades[1, 0] = 85;
            studentGrades[1, 1] = 86;
            studentGrades[1, 2] = 83;
            studentGrades[1, 3] = 90;

            //sally grades
            studentGrades[2, 0] = 99;
            studentGrades[2, 1] = 98;
            studentGrades[2, 2] = 100;
            studentGrades[2, 3] = 97;

            // jake grades
            studentGrades[3, 0] = 95;
            studentGrades[3, 1] = 82;
            studentGrades[3, 2] = 84;
            studentGrades[3, 3] = 90;

            // amy grades
            studentGrades[4, 0] = 98;
            studentGrades[4, 1] = 89;
            studentGrades[4, 2] = 86;
            studentGrades[4, 3] = 99;

            //fred average
            studentAverage[0] =(decimal) (studentGrades[0, 0] + studentGrades[0, 1] + studentGrades[0, 2] + studentGrades[0, 3]) / 4;

            //barbara average
            studentAverage[1] = (decimal) (studentGrades[1, 0] + studentGrades[1, 1] + studentGrades[1, 2] + studentGrades[1, 3]) / 4;

            //sally average
            studentAverage[2] = (decimal) (studentGrades[2, 0] + studentGrades[2, 1] + studentGrades[2, 2] + studentGrades[2, 3]) / 4;

            //jake average
            studentAverage[3] =(decimal) (studentGrades[3, 0] + studentGrades[3, 1] + studentGrades[3, 2] + studentGrades[3, 3]) / 4;

            //amy avaerage
            studentAverage[4] = (decimal) (studentGrades[4, 0] + studentGrades[4, 1] + studentGrades[4, 2] + studentGrades[4, 3]) / 4;
        }
    }
}
